def conn
  @conn ||= Sequel.sqlite(
    Config.db_path,
    timeout:      10_000,
    journal_mode: :wal,
    synchronous:  :normal
  )
end

def load_schema!
  path = File.expand_path('db_schema.sql', DATA_DIR)
  return unless conn.tables.empty?
  conn.run(File.read(path))
  Log.log(:info, :DB, "Schema initialized -> #{path}")
  conn.run 'PRAGMA foreign_keys = ON'
end
